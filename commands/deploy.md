### Fastfile

```ruby
# fastlane/Fastfile
# Generated by SwiftShip /apple:deploy

default_platform(:ios)

platform :ios do

  # ═══════════════════════════════════════════════════════════
  # SETUP & SIGNING
  # ═══════════════════════════════════════════════════════════

  desc "Sync certificates and profiles"
  lane :sync_signing do
    match(type: "development")
    match(type: "appstore")
  end

  # ═══════════════════════════════════════════════════════════
  # TESTING
  # ═══════════════════════════════════════════════════════════

  desc "Run tests"
  lane :test do
    run_tests(
      scheme: "[SCHEME_NAME]",
      device: "iPhone 15 Pro",
      clean: true
    )
  end

  # ═══════════════════════════════════════════════════════════
  # TESTFLIGHT (BETA)
  # ═══════════════════════════════════════════════════════════

  desc "Push a new beta build to TestFlight"
  lane :beta do
    # Ensure we're on a clean state
    ensure_git_status_clean

    # Increment build number
    increment_build_number(
      build_number: latest_testflight_build_number + 1
    )

    # Sync signing
    match(type: "appstore", readonly: true)

    # Build
    build_app(
      scheme: "[SCHEME_NAME]",
      export_method: "app-store",
      clean: true
    )

    # Upload to TestFlight
    upload_to_testflight(
      skip_waiting_for_build_processing: true,
      changelog: git_changelog
    )

    # Commit build number change
    commit_version_bump(
      message: "Build #{lane_context[SharedValues::BUILD_NUMBER]}",
      force: true
    )

    # Tag the release
    add_git_tag(tag: "beta/#{lane_context[SharedValues::BUILD_NUMBER]}")

    # Push to remote
    push_to_git_remote

    # Notify (optional)
    # slack(message: "New beta uploaded! 🚀")
  end

  # ═══════════════════════════════════════════════════════════
  # APP STORE (RELEASE)
  # ═══════════════════════════════════════════════════════════

  desc "Push a new release to the App Store"
  lane :release do
    # Ensure clean state
    ensure_git_status_clean

    # Bump version (patch by default)
    increment_version_number(bump_type: "patch")
    increment_build_number(build_number: 1)

    # Sync signing
    match(type: "appstore", readonly: true)

    # Build
    build_app(
      scheme: "[SCHEME_NAME]",
      export_method: "app-store",
      clean: true
    )

    # Upload to App Store
    upload_to_app_store(
      skip_metadata: false,
      skip_screenshots: true,  # Set false if you have screenshots
      submit_for_review: false,  # Set true for auto-submit
      automatic_release: false,
      force: true
    )

    # Commit and tag
    version = get_version_number
    commit_version_bump(message: "Release #{version}")
    add_git_tag(tag: "v#{version}")
    push_to_git_remote
  end

  # ═══════════════════════════════════════════════════════════
  # SCREENSHOTS
  # ═══════════════════════════════════════════════════════════

  desc "Capture screenshots"
  lane :screenshots do
    capture_screenshots(
      scheme: "[SCHEME_NAME]UITests",
      output_directory: "./fastlane/screenshots",
      clear_previous_screenshots: true
    )
    frame_screenshots(white: true)
  end

  # ═══════════════════════════════════════════════════════════
  # UTILITIES
  # ═══════════════════════════════════════════════════════════

  desc "Register new device"
  lane :register_device do
    device_name = prompt(text: "Device name: ")
    device_udid = prompt(text: "Device UDID: ")
    register_devices(devices: { device_name => device_udid })
    match(type: "development", force_for_new_devices: true)
  end

  # Error handling
  error do |lane, exception|
    # slack(
    #   message: "Error in #{lane}: #{exception.message}",
    #   success: false
    # )
  end
end

# ═══════════════════════════════════════════════════════════
# macOS PLATFORM (if applicable)
# ═══════════════════════════════════════════════════════════

# platform :mac do

#     # Similar to iOS but for macOS
#   end
# end
```

## Step 5: Generate GitHub Actions (if selected)

### Build & Test Workflow

```yaml
# .github/workflows/build-test.yml
# Generated by SwiftShip /apple:deploy

name: Build & Test

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  build:
    runs-on: macos-26   # latest stable runner — see github.com/actions/runner-images

    steps:
      - uses: actions/checkout@v4

      - name: Select Xcode (latest stable)
        uses: maxim-lobanov/setup-xcode@v1
        with:
          xcode-version: latest-stable   # currently Xcode 26.x / Swift 6.2 — auto-tracks newest

      - name: Cache SPM
        uses: actions/cache@v3
        with:
          path: .build
          key: ${{ runner.os }}-spm-${{ hashFiles('**/Package.resolved') }}
          restore-keys: ${{ runner.os }}-spm-

      - name: Build
        run: |
          xcodebuild build \
            -scheme "[SCHEME_NAME]" \
            -destination "generic/platform=iOS Simulator" \
            -skipPackagePluginValidation \
            CODE_SIGNING_ALLOWED=NO

      - name: Test
        run: |
          xcodebuild test \
            -scheme "[SCHEME_NAME]" \
            -destination "platform=iOS Simulator,name=iPhone 17 Pro,OS=latest" \
            -skipPackagePluginValidation \
            CODE_SIGNING_ALLOWED=NO
```

### TestFlight Deployment Workflow

```yaml
# .github/workflows/deploy-testflight.yml
# Generated by SwiftShip /apple:deploy

name: Deploy to TestFlight

on:
  push:
    branches: [main]
    # Or use tags: tags: ['beta/*']

jobs:
  deploy:
    runs-on: macos-26   # latest stable runner — see github.com/actions/runner-images

    steps:
      - uses: actions/checkout@v4

      - name: Select Xcode (latest stable)
        uses: maxim-lobanov/setup-xcode@v1
        with:
          xcode-version: latest-stable   # currently Xcode 26.x / Swift 6.2 — auto-tracks newest

      - name: Install fastlane
        run: brew install fastlane

      - name: Setup signing
        env:
          MATCH_PASSWORD: ${{ secrets.MATCH_PASSWORD }}
          MATCH_GIT_BASIC_AUTHORIZATION: ${{ secrets.MATCH_GIT_AUTH }}
        run: fastlane sync_signing

      - name: Build & Upload
        env:
          APP_STORE_CONNECT_API_KEY_ID: ${{ secrets.ASC_KEY_ID }}
          APP_STORE_CONNECT_API_ISSUER_ID: ${{ secrets.ASC_ISSUER_ID }}
          APP_STORE_CONNECT_API_KEY_CONTENT: ${{ secrets.ASC_KEY_CONTENT }}
        run: fastlane beta
```

## Initial Setup

### 1. Install Fastlane
```bash
brew install fastlane
```

### 2. Set Up Code Signing (match)
```bash
# Create private repo for certificates
# Then initialize match:
fastlane match init

# Generate certificates:
fastlane match development
fastlane match appstore
```

### 3. Create App Store Connect API Key
1. Go to App Store Connect → Users and Access → Keys
2. Generate new key with "App Manager" role
3. Download .p8 file (only available once!)
4. Note Key ID and Issuer ID

### 4. Configure GitHub Secrets
Add these secrets to your repo (Settings → Secrets):

| Secret | Value |
|--------|-------|
| `ASC_KEY_ID` | App Store Connect Key ID |
| `ASC_ISSUER_ID` | App Store Connect Issuer ID |
| `ASC_KEY_CONTENT` | Contents of .p8 file |
| `MATCH_PASSWORD` | Password for match encryption |
| `MATCH_GIT_AUTH` | Base64 encoded git credentials |

# Reset and re-sync certificates

fastlane match nuke distribution
fastlane match development
fastlane match appstore
```

```bash
# Clean build
rm -rf ~/Library/Developer/Xcode/DerivedData
fastlane beta
