## Step 4: Generate UI Test for Automation

Create a UI test file that captures screenshots using Xcode's screenshot API.

### Find UI Test Target

```bash
# Look for existing UI test files
find . -name "*UITests*.swift" -o -name "*UITest*.swift" | head -5
```

### Generate Screenshot Test

Create/update UI test file:

```swift
// ScreenshotTests.swift
// Generated by SwiftShip /apple:screenshots

import XCTest

final class ScreenshotTests: XCTestCase {

    let app = XCUIApplication()

    override func setUpWithError() throws {
        continueAfterFailure = false
        app.launchArguments = ["--uitesting", "--screenshots"]

        // Set language for localized screenshots
        app.launchArguments += ["-AppleLanguages", "(en)"]
        app.launchArguments += ["-AppleLocale", "en_US"]

        app.launch()
    }

    // MARK: - Screenshot Capture

    func testCaptureScreenshots() throws {
        // Screenshot 1: Hero - Main screen
        // TODO: Navigate to main screen if needed
        takeScreenshot(named: "01_Hero")

        // Screenshot 2: Core Feature
        // TODO: Interact to show core feature
        // app.buttons["StartButton"].tap()
        takeScreenshot(named: "02_CoreFeature")

        // Screenshot 3: Differentiator
        // TODO: Navigate to unique feature
        takeScreenshot(named: "03_Differentiator")

        // Screenshot 4: Secondary Feature
        // TODO: Navigate to secondary feature
        takeScreenshot(named: "04_SecondaryFeature")

        // Screenshot 5: Settings/Trust
        // TODO: Navigate to settings or final screen
        takeScreenshot(named: "05_Trust")
    }

    // MARK: - Dark Mode Variants

    func testCaptureScreenshotsDarkMode() throws {
        // Note: Dark mode screenshots require device/simulator setting
        // Or use: XCUIDevice.shared.appearance = .dark (iOS 13+)

        takeScreenshot(named: "01_Hero_Dark")
        // ... repeat for other screenshots
    }

    // MARK: - Helper

    private func takeScreenshot(named name: String) {
        let screenshot = XCUIScreen.main.screenshot()
        let attachment = XCTAttachment(screenshot: screenshot)
        attachment.name = name
        attachment.lifetime = .keepAlways
        add(attachment)
    }
}
```

## Step 5: Generate fastlane snapshot Configuration

If using fastlane for screenshots:

### Snapfile

```ruby
# fastlane/Snapfile
# Generated by SwiftShip /apple:screenshots

# Devices to capture
devices([
  "iPhone 15 Pro Max",
  "iPhone 8 Plus",
  # "iPad Pro (12.9-inch) (6th generation)"  # Uncomment if universal
])

# Languages
languages([
  "en-US",
  # "de-DE",  # Add more as needed
  # "ja",
])

# Scheme containing UI tests
scheme("[APP_SCHEME]UITests")

# Output directory
output_directory("./fastlane/screenshots")

# Clear previous screenshots
clear_previous_screenshots(true)

# Stop on first error
stop_after_first_error(true)

# Don't reinstall app
reinstall_app(false)

# Use dark mode
# dark_mode(true)  # Uncomment for dark mode set
```

## Capture Instructions

### Manual Capture

2. Set up each screen state
3. Cmd+S to save screenshot
4. Process with design tool

### Automated Capture (fastlane)
```bash
fastlane snapshot
fastlane frameit
```

### Automated Capture (Xcode)
```bash
xcodebuild test \
  -scheme "[Scheme]UITests" \
  -destination "platform=iOS Simulator,name=iPhone 15 Pro Max" \
  -testPlan Screenshots
